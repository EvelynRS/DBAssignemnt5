CREATE TABLE team(
TEAM_ID int,
TEAM_NAME char(50) NOT NULL,
NICKNAME char(50) NOT NULL,
RANK int NOT NULL,
PRIMARY KEY (TEAM_ID)
);

CREATE TABLE game(
GAME_ID int,
RANK1 char(50) NOT NULL,
RANK2 char(50) NOT NULL,
LOCATION char(50) NOT NULL,
DATE date NOT NULL,
PRIMARY KEY (GAME_ID)
);

CREATE TABLE result(
GAME_ID int,
TEAM_ONE_ID char(50) NOT NULL,
TEAM_TWO_ID char(50) NOT NULL,
SCORE_ONE int NOT NULL,
SCORE_TWO int NOT NULL,
FOREIGN KEY (GAME_ID) REFERENCES game(GAME_ID)
);
